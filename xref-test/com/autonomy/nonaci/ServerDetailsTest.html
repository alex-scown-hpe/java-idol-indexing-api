228

230         assertThat("Details should be equal to itself", details1.equals(details1), is(true));
231         assertThat("Details should not be equal", details1.equals(details2), is(false));
232         
233         // Change details2 to have the same name so technically they return the same...
234         details2.setHost("host1.example.com");
235         assertThat("Details should be equal", details1.equals(details2), is(true));
236         
237         // Compare it to some random object, it should be false...
238         assertThat("Parameters should be equal", details1.equals(new Object()), is(false));

243         details3.setPort(5790);
244         details3.setCharsetName("UTF-8");
