22  /**
23   * This test class is just to check that a sub class of IndexCommandImpl works if changes to that class have taken
24   * place...
25   *
26   * @author boba
27   */
28  public class DreReplaceCommandTest {
29  
30      private static String queryString;
31  
32      private DreReplaceCommand command;
33  
34      @BeforeClass
35      public static void createQueryString() {
36          queryString = IndexCommand.PARAM_DATABASE_MATCH + "=test&"
37              + IndexCommand.PARAM_MULTIPLE_VALUES + "=true&"
38              + IndexCommand.PARAM_REPLACE_ALL_REFS + "=true";
39      }
40  
41  
42      @Before
43      public void createCommand() {
44          // The actual command doesn't really matter...
45          command = new DreReplaceCommand();
46      }
47  
48      @Test
49      @SuppressWarnings("unchecked")
50      public void testParameterMapMethods() {
51          // Parameters should be empty...
52          assertThat("parameters", command.getParameters().isEmpty(), is(true));
53  
54          // Set some parameters...
55          final Map<String, String> parameters = new LinkedHashMap<String, String>();
56          parameters.put(IndexCommand.PARAM_DATABASE_MATCH, "test");
57          parameters.put(IndexCommand.PARAM_MULTIPLE_VALUES, "true");
58          parameters.put(IndexCommand.PARAM_REPLACE_ALL_REFS, "true");
59          command.setParameters(parameters);
60  
61          // Parameters should no longer be empty and the returned map should be a different instance...
62          assertThat("parameters", command.getParameters().size(), is(3));
63          assertThat("parameters", command.getParameters(), is(not(sameInstance(parameters))));
64          assertThat("queryString", command.getQueryString(), is(equalTo(queryString)));
65  
66          // I should be able to modify our copy of the map and the command's copy shouldn't change...
67          parameters.put(IndexCommand.PARAM_BLOCKING, "true");
68          assertThat("parameters", command.getParameters().size(), is(3));
69          assertThat("queryString", command.getQueryString(), is(equalTo(queryString)));
70      }
71  
72      @Test
73      @SuppressWarnings("unchecked")
74      public void testParameterMethods() {
75          // Parameters should be empty...
76          assertThat("parameters", command.getParameters().isEmpty(), is(true));
77  
78          // Set some parameters...
79          command.setDatabaseMatch("test");
80          command.setMultipleValues(true);
81          command.setReplaceAllRefs(true);
82  
83          // Parameters should no longer be empty...
84          assertThat("parameters", command.getParameters().size(), is(3));
85          assertThat("queryString", command.getQueryString(), is(equalTo(queryString)));
86      }
