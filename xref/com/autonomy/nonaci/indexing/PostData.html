5   package com.autonomy.nonaci.indexing;
6   
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.io.OutputStream;
10  
11  /**
12   * Defines methods for handling post data. This interface is based on the Apache HttpClient <tt>HttpEntity</tt> 
13   * interface, but without any reference to other HttpClient specific classes or interfaces.
14   *
15   * @author boba
16   */
17  public interface PostData {
18  
19      /**
20       * Tells if the entity is capable of producing its data more than once. A repeatable entity's getContent() and
21       * writeTo(OutputStream) methods can be called more than once whereas a non-repeatable entity's can not.
22       *
23       * @return <tt>true</tt> if the entity is repeatable, <tt>false</tt> otherwise.
24       */
25      boolean isRepeatable();
26  
27      /**
28       * Tells about chunked encoding for this entity. The primary purpose of this method is to indicate whether chunked
29       * encoding should be used when the entity is sent.
30       *
31       * @return <tt>true</tt> if chunked encoding is preferred for this entity, or <tt>false</tt> if it is not
32       */
33      boolean isChunked();
34  
35      /**
36       * Tells the length of the content, if known.
37       *
38       * @return the number of bytes of the content, or a negative number if unknown. If the content length is known but
39       * exceeds {@link java.lang.Long#MAX_VALUE Long.MAX_VALUE}, a negative number is returned.
40       */
41      long getContentLength();
42  
43      /**
44       * Obtains the Content-Type header, if known. This is the header that should be used when sending the entity, It can
45       * include a charset attribute.
46       *
47       * @return the Content-Type header for this entity, or <tt>null</tt> if the content type is unknown
48       */
49      String getContentType();
50  
51      /**
52       * Obtains the Content-Encoding header, if known. This is the header that should be used when sending the entity.
53       * Wrapping entities that modify the content encoding should adjust this header accordingly.
54       *
55       * @return the Content-Encoding header for this entity, or <tt>null</tt> if the content encoding is unknown
56       */
57      String getContentEncoding();
58  
59      /**
60       * Creates a new InputStream object of the entity. It is a programming error to return the same InputStream object
61       * more than once. Entities that are not {@link #isRepeatable repeatable} will throw an exception if this method is
62       * called multiple times.
63       *
64       * @return a new input stream that returns the entity data.
65       * @throws IOException if the stream could not be created
66       * @throws IllegalStateException if this entity is not repeatable and the stream has already been obtained
67       * previously
68       */
69      InputStream getContent() throws IOException;
70  
71      /**
72       * Writes the entity content to the output stream.
73       *
74       * @param outstream the output stream to write entity content to
75       * @throws IOException if an I/O error occurs
76       */
77      void writeTo(OutputStream outstream) throws IOException;
78  
79      /**
80       * Tells whether this entity depends on an underlying stream. Streamed entities should return <tt>true</tt> until
81       * the content has been consumed, <tt>false</tt> afterwards. Self-contained entities should return <tt>false</tt>.
82       * Wrapping entities should delegate this call to the wrapped entity.
83       * <p />
84       * The content of a streamed entity is consumed when the stream returned by {@link #getContent getContent} has been
85       * read to EOF, or after {@link #finish() finish} has been called. If a streamed entity can not detect whether the
86       * stream has been read to EOF, it should return <tt>true</tt> until {@link #finish() finish} is called.
87       *
88       * @return <tt>true</tt> if the entity content is streamed and not yet consumed, <tt>false</tt> otherwise
89       */
90      boolean isStreaming();
91  
92      /**
93       * This method is called to indicate that the content of this entity is no longer required. All entity
94       * implementations are expected to release all allocated resources as a result of this method invocation. Content
95       * streaming entities are also expected to dispose of the remaining content, if any. Wrapping entities should
96       * delegate this call to the wrapped entity.
97       *
98       * @throws IOException if an I/O error occurs. This indicates that connection keep-alive is not possible.
99       */
100     void finish() throws IOException;
101 
102 }
