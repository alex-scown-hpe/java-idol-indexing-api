108     @Override
109     public String getQueryString() {
110         final List<NameValuePair> pairs = new ArrayList<NameValuePair>(parameters.size());
111 
112         for(final Map.Entry<String, String> entry : parameters.entrySet()) {
113             pairs.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
114         }
115 
116         return URLEncodedUtils.format(pairs, "UTF-8");

144     /**
