32  /**
33   * Core implementation of the {@link com.autonomy.nonaci.indexing.IndexingService} interface. This implementation uses
34   * the Apache {@link org.apache.http.client.HttpClient} library to do the actual communication with the IDOL's index
35   * port.
36   * <p/>
37   * The {@link org.apache.http.client.HttpClient} instance to be used should be set before calling either of the execute
38   * methods. The {@link com.autonomy.nonaci.ServerDetails} can either be set before calling the execute method, or
39   * optionally sent as a method parameter to the execute method. This allows the API to be set up to talk to a single IDOL
40   * server by setting the details once, or multiple IDOL Servers by sending the details with every invocation of execute.

68      private URI createIndexCommandURI(final ServerDetails serverDetails, final IndexCommand command) throws URISyntaxException {
69          LOGGER.trace("createIndexCommandURI() called...");
70  
71          // Create the URI to use... We can't use URLEncodedUtils.parse() or setQuery() since they decode
72          // command.getQueryString(), which we shouldn't do since both '+' and '%20' will be decoded-and-reencoded
73          // to '+', which causes problems since IDOL treats these differently e.g. when listing DREREFERENCEs.
74          final URI uri = new URIBuilder()
75                  .setScheme(serverDetails.getProtocol().toString().toLowerCase(Locale.ENGLISH))
76                  .setHost(serverDetails.getHost())
77                  .setPort(serverDetails.getPort())
78                  .setPath('/' + command.getCommand())
79                  .build();
80  
81          final String qs = command.getQueryString();
82  
83          return StringUtils.isBlank(qs) ? uri : new URI(uri + "?" + qs);
84      }

93      private HttpUriRequest createGetMethod(final ServerDetails serverDetails, final IndexCommand command) throws URISyntaxException {
94          LOGGER.trace("createGetMethod() called...");
95  
96          // Return the constructed get method...
97          return new HttpGet(createIndexCommandURI(serverDetails, command));
98      }

100     /**
101      * Create a <tt>HttpClient</tt> <tt>PostMethod</tt> instance that contains the POST content.
102      *
103      * @param serverDetails The connection details of the server to use...
104      * @param command The command to get the POST content from
105      * @return a <tt>HttpUriRequest</tt> containing the contents of the <tt>Command</tt>
106      * @throws java.net.URISyntaxException If there was a problem converting the IndexCommand into a URI
107      */
108     private HttpUriRequest createPostMethod(final ServerDetails serverDetails, final IndexCommand command) throws URISyntaxException {
109         LOGGER.trace("createPostMethod() called...");
110 
111         final HttpPost httpPost = new HttpPost(createIndexCommandURI(serverDetails, command));
112         httpPost.setEntity(new PostDataHttpEntity(command.getPostData()));
113         return httpPost;
114     }

131     /**
132      * Executes an index command
133      *
134      * @param serverDetails The connection details of the indexing server.
135      * @param command The index command to execute
136      * @return the index queue id for the command
137      * @throws com.autonomy.nonaci.indexing.IndexingException if an error response was detected, or if something went
138      * wrong while executing the command
139      */
140     @Override
141     public int executeCommand(final ServerDetails serverDetails, final IndexCommand command) throws IndexingException {
142         LOGGER.trace("executeCommand() called...");
143 
144         // Sanity check...
145         Validate.notNull(serverDetails, "Indexing server connection details must be set before calling this method.");
146         Validate.notNull(command.getCommand(), "You must set a valid index command to execute");
147         Validate.notNull(httpClient, "You must set httpClient before executing commands");
148 
149         HttpResponse httpResponse = null;
150 
151         try {
152             // Create the HTTP method to use...
153             final HttpUriRequest httpRequest = (command.getPostData() == null)
154                     ? createGetMethod(serverDetails, command)
155                     : createPostMethod(serverDetails, command);
156 
157             LOGGER.debug("Executing a {} index command...", command.getCommand());
158 
159             // Execute the command...
160             httpResponse = httpClient.execute(httpRequest);
161 
162             LOGGER.debug("Successfully executed the index command, HTTP status - {}...", httpResponse.getStatusLine().getStatusCode());
163 
164             // Get the response...
165             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
166             httpResponse.getEntity().writeTo(baos);
167             final String response = baos.toString("UTF-8");
168 
169             if(!response.contains("INDEXID")) {
170                 throw new IndexingException(response.trim());
171             }
172 
173             // Get the index queue id...
174             final Matcher matcher = pattern.matcher(response);
175             if(matcher.matches()) {
176                 return NumberUtils.toInt(matcher.group(1));
177             }
178 
179             throw new IndexingException("Unable to get index queue id from response: [" + response + ']');
180         }
181         catch(final IOException ioe) {
182             throw new IndexingException("Unable to execute the Index command", ioe);
183         }
184         catch(final URISyntaxException urise) {
185             throw new IndexingException("Unable to construct the Index command URI.", urise);
186         }
187         finally {
188             try {
189                 if(httpResponse != null) {
190                     // Release the HTTP connection...
191                     EntityUtils.consume(httpResponse.getEntity());
192                 }
193             }
194             catch(final IOException ioe) {
195                 LOGGER.error("Unable to consume any remaining content from the IndexCommand response.", ioe);
196             }
197         }
198     }
