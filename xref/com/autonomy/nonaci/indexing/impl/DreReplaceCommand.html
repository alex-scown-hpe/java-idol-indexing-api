9   /**
10   * <strong>Note:</strong> This command requires a POST request method.
11   * <p />
12   * Use the <strong>DREREPLACE</strong> command to change the field values or delete fields from documents.
13   *
14   * @author boba
15   */
16  public class DreReplaceCommand extends IndexCommandImpl {
17  
18      /** Creates a new instance of <tt>DreReplaceCommand</tt>. */
19      public DreReplaceCommand() {
20          super(CMD_DREREPLACE);
21      }
22  
23      public String getDatabaseMatch() {
24          return get(PARAM_DATABASE_MATCH);
25      }
26  
27      /**
28       * Enter the databases in which you want to replace the specified data. If you want to specify multiple databases,
29       * you must separate them with plus symbols (there must be no space before or after a plus symbol).
30       *
31       * @param databaseMatch The databases in which data is to be replaced.
32       */
33      public void setDatabaseMatch(final String databaseMatch) {
34          put(PARAM_DATABASE_MATCH, databaseMatch);
35      }
36  
37      public boolean isInsertValue() {
38          return BooleanUtils.toBoolean(get(PARAM_INSERT_VALUE));
39      }
40  
41      /**
42       * If you specify <strong>true</strong>, the specified field and value are simply added to the specified document.
43       * If there are existing instances of the field, the data is added as a new instance. If you specify <strong>false
44       * </strong>, all fields whose name matches the specified field are removed from the document before the specified
45       * field and value are written into it. The specified data replaces any pre-existing data for that field.
46       *
47       * @param insertValue Whether to add new instances of a specified field, or just replace values in existing
48       *         instances.
49       */
50      public void setInsertValue(final boolean insertValue) {
51          put(PARAM_INSERT_VALUE, BooleanUtils.toStringTrueFalse(insertValue));
52      }
53  
54      public String getField() {
55          return get(PARAM_FIELD);
56      }
57  
58      /**
59       * The field containing the reference specified by #DREDOCREF when a document has more than one reference.
60       * <p />
61       * You can specify multiple fields by separating them with commas or spaces (there must be no space before or after 
62       * a comma).
63       * 
64       * @param field The field containing the reference specified by #DREDOCREF.  	
65       */
66      public void setField(final String field) {
67          put(PARAM_FIELD, field);
68      }
69  
70      public boolean isMultipleValues() {
71          return BooleanUtils.toBoolean(get(PARAM_MULTIPLE_VALUES));
72      }
73  
74      /**
75       * Specify <strong>true</strong> to allow the data to contain multiple #DREFIELDNAME/#DREFIELDVALUE pairs in which
76       * #DREFIELDNAME has the same value. This allows you to create multiple instances of the same field with different
77       * values. Specify <strong>false</strong> to require that the data contain only one #DREFIELDNAME/#DREFIELDVALUE
78       * pair for each field name.
79       *
80       * @param multipleValues Whether to allow multiple #DREFIELDNAME/#DREFIELDVALUE pairs for the same field name.
81       */
82      public void setMultipleValues(final boolean multipleValues) {
83          put(PARAM_MULTIPLE_VALUES, BooleanUtils.toStringTrueFalse(multipleValues));
84      }
85  
86      public boolean isReplaceAllRefs() {
87          return BooleanUtils.toBoolean(get(PARAM_REPLACE_ALL_REFS));
88      }
89  
90      /**
91       * If you specify <strong>true</strong>, and if you use #DREDOCREF in the replacement data to identify documents,
92       * the <strong>DREREPLACE</strong> command will affect all documents with a reference field whose value matches the
93       * value supplied in #DREDOCREF.
94       * <p />
95       * If you specify <strong>false</strong>, the command will affect only the <em>first</em> matching document.
96       *
97       * @param replaceAllRefs Whether to replace values in all matching documents, or just the first one.
98       */
99      public void setReplaceAllRefs(final boolean replaceAllRefs) {
100         put(PARAM_REPLACE_ALL_REFS, BooleanUtils.toStringTrueFalse(replaceAllRefs));
101     }
