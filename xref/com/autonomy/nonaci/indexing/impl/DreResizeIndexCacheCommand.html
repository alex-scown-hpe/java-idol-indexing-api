7   import org.apache.commons.lang3.math.NumberUtils;
8   
9   /**
10   * Allows you to resize the IDOL server indexing cache without restarting the server.
11   * The default size of the index cache at IDOL server startup is determined by the value of the <strong>
12   * IndexCacheMaxSize</strong> configuration parameter in the <strong>[IndexCache]</strong> section of the IDOL server 
13   * configuration file. Using the <strong>DRERESIZEINDEXCACHE</strong> command allows you to change the cache size 
14   * dynamically -- for example, to decrease it after performing a large bulk indexing job.
15   * <p />
16   * <strong>Note:</strong> Because the cache must be flushed before it is resized, invoking this command more often than 
17   * necessary can degrade indexing performance.
18   *
19   * @author boba
20   */
21  public class DreResizeIndexCacheCommand extends IndexCommandImpl {
22  
23      /** Creates a new instance of <tt>DreResizeIndexCacheCommand</tt>. */
24      public DreResizeIndexCacheCommand() {
25          super(CMD_DRERESIZEINDEXCACHE);
26      }
27  
28      /**
29       * @return If the return value is <tt>-1</tt> this indicates that the property hasn't been set.
30       */
31      public int getIndexCacheMaxSize() {
32          return NumberUtils.toInt(get(PARAM_INDEX_CACHE_MAX_SIZE), -1);
33      }
34  
35      /**
36       * Specify the desired size (in Kb) for the cache index. If the full requested amount is not available, the system 
37       * will attempt to allocate as much of it as possible.
38       * 
39       * @param indexCacheMaxSize The new cache size.
40       * @throws java.lang.IllegalArgumentException If the specified value is out of range (10240 < <tt>indexCacheMaxSize
41       * </tt> < 2048000).
42       */
43      public void setIndexCacheMaxSize(final int indexCacheMaxSize) {
44          if((indexCacheMaxSize < 10240) || (indexCacheMaxSize > 2048000)) {
45              throw new IllegalArgumentException("The specified indexCacheMaxSize is out of range. The allowed range is 10240 < indexCacheMaxSize < 2048000.");
46          }
47          
48          put(PARAM_INDEX_CACHE_MAX_SIZE, String.valueOf(indexCacheMaxSize));
49      }
